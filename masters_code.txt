char ToLower(char c)
{
    if ((c >= 'A') && (c <= 'Z')) return (char)((int)c - 'A' + 'a');
    else return c;
}

int Vowel(char c)
{
    c = ToLower(c);
    return (c == 'a') || (c == 'o') || (c == 'i') || (c == 'y') || (c == 'u') || (c == 'e');
}

void Solve()
{
    string t = "";
    int i;
    for (i = 0; i < s.length(); i++)
    {
        if (Vowel(s[i])) continue;
        t += ".";
        t += ToLower(s[i]);
    }
    cout << t;
}

int main()
{
    Load();
    Solve();
    return 0;
}
